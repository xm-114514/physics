// #1
const g = 9.8; // 重力加速度 (m/s^2)
const v0 = 10; // 初速度 (m/s)
const angle = Math.PI / 4; // 発射角度 (45度)
  
// 2
const velocityX = v0 * Math.cos(angle); // 水平方向の速度
const xPosition = velocityX * t; // 水平方向の位置
ball.position.x = xPosition; // ボールの水平方向の位置更新

// 3.
const velocityY = v0 * Math.sin(angle); // 鉛直方向の速度
const yPosition = velocityY * t - 0.5 * g * t ** 2; // 鉛直方向の位置
ball.position.y = yPosition; // ボールの鉛直方向の位置更新

// 4.
const velocity = {
  x: v0 * Math.cos(angle), // 水平方向の初速度
  y: v0 * Math.sin(angle), // 鉛直方向の初速度
};
// 5.
// 瞬間速度の大きさ
const velocityMagnitude = Math.sqrt(velocity.x ** 2 + (velocity.y - g * t) ** 2);

// 6.
const mass = 1; // ボールの質量 (kg)
const kineticEnergy = 0.5 * mass * velocityMagnitude ** 2; // 運動エネルギー
const potentialEnergy = mass * g * ball.position.y; // 位置エネルギー

#1 初期条件 \( g = 9.8 \, \mathrm{m/s^2}, \quad v_0 = 10 \, \mathrm{m/s}, \quad \theta = \frac{\pi}{4} \)
#2 水平方向の位置 \( x(t) = v_0 \cos(\theta) \cdot t \)
#3 鉛直方向の位置 \( y(t) = v_0 \sin(\theta) \cdot t - \frac{1}{2} g t^2 \)
#4 速度の分解 \( v_x = v_0 \cos(\theta), \quad v_y = v_0 \sin(\theta) \)
#5 瞬間速度の大きさ \( v(t) = \sqrt{v_x^2 + (v_y - g \cdot t)^2} \)
#6 エネルギー \( K = \frac{1}{2} m v^2, \quad U = m g y \)
